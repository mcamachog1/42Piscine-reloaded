COLOCAR PRIVADO UN REPOSITORIO
curl \
  -X PATCH \
  -H "Accept: application/vnd.github+json" \
  -H "Authorization: Bearer token " \
  https://api.github.com/repos/mcamachog1/onlydepas \
  -d '{"private":true}'

OBTENER UN LISTADO DE TODOS MIS REPOSITORIOS
curl -L \
  -H "Accept: application/vnd.github+json" \
  -H "Authorization: Bearer token " \
  https://api.github.com/users/mcamachog1/repos | jq -r '.[].name'

OBTENER UN LISTADO DE LOS REPOSITORIOS PRIVADOS
curl -L \
  -H "Accept: application/vnd.github+json" \
  -H "Authorization: Bearer token " \
  https://api.github.com/user/repos | jq -r '.[] | select(.private == true) | .name'

OBTENER UN LISTADO DE LOS REPOSITORIOS PUBLICOS
curl -L \
  -H "Accept: application/vnd.github+json" \
  -H "Authorization: Bearer token " \
  https://api.github.com/user/repos | jq -r '.[] | select(.private == false) | .name'
